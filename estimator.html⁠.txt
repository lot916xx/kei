<script>
function calculateEstimation() {
    // 入力値の取得
    const punching = parseFloat(document.getElementById('punching').value) || 0;
    const stitches = parseFloat(document.getElementById('stitches').value) || 0;
    const ratePerStitch = parseFloat(document.getElementById('ratePerStitch').value) || 0;
    const quantity = parseInt(document.getElementById('quantity').value) || 0;
    const bodyCost = parseFloat(document.getElementById('bodyCost').value) || 0;

    if (quantity <= 0) {
        alert("数量は1以上にしてください。");
        return;
    }

    // 計算
    // 1枚あたりの刺繍代 = (針数 / 1000) * 1000針単価
    const embroideryPerUnit = (stitches / 1000) * ratePerStitch;
    const embroideryTotal = embroideryPerUnit * quantity;
    const bodyTotal = bodyCost * quantity;
    
    // 総合計 = 型代 + 刺繍代合計 + ボディ代合計 (今回は簡易的に10%の消費税をプラス)
    const subTotal = punching + embroideryTotal + bodyTotal;
    const grandTotal = Math.floor(subTotal * 1.1); // 税込み（端数切り捨て）
    
    const perUnitTotal = Math.floor(grandTotal / quantity);

    // 画面に反映
    document.getElementById('resPunching').innerText = punching.toLocaleString() + " 円";
    document.getElementById('resEmbroideryPerUnit').innerText = embroideryPerUnit.toLocaleString() + " 円";
    document.getElementById('resEmbroideryTotal').innerText = embroideryTotal.toLocaleString() + " 円";
    document.getElementById('resBodyTotal').innerText = bodyTotal.toLocaleString() + " 円";
    document.getElementById('resGrandTotal').innerText = grandTotal.toLocaleString() + " 円";
    document.getElementById('resPerUnit').innerText = perUnitTotal.toLocaleString() + " 円";

    // 結果ボックスを表示
    document.getElementById('resultBox').style.display = 'block';
}
</script>
